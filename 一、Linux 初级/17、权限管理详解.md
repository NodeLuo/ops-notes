
1、批量创建用户

```bash
1.带空格拼接
	[root@breadbomb ~]# echo 'useradd bob'{1..3}|xargs -n 2
	useradd bob1
	useradd bob2
	useradd bob3

2.分隔符拼接
	[root@breadbomb ~]# echo 'useradd_bob'{1..3}|sed 's#_# #g'|xargs -n 2
	useradd bob1
	useradd bob2
	useradd bob3

3.后向引用拼接
	[root@breadbomb ~]# echo 'bob'{1..3}|xargs -n 1|sed -E 's#(.*)#useradd \1#g'
	useradd bob1
	useradd bob2
	useradd bob3

#以上最后统一加上 |bash 即可执行

4.批量创建用户并设置密码
	[root@breadbomb ~]# echo 'bob'{1..3}|xargs -n1|sed -E 's#(.*)#useradd \1;echo 123456|passwd --stdin \1#g'
	useradd bob1;echo 123456|passwd --stdin bob1
	useradd bob2;echo 123456|passwd --stdin bob2
	useradd bob3;echo 123456|passwd --stdin bob3

5.使用mkpasswd设置密码
	echo 'bob'{1..3}|xargs -n1|sed -E 's#(.*)#useradd \1;echo `mkpasswd`|passwd --stdin \1#g'

6.存放密码操作
	|tee -a password.txt #将这一串拼接在上面，相当于结果留一份，文件留一份
	
	tee    是默认清空后存入
	tee -a 是追加操作

7.存放用户名+密码操作
	echo 'bob'{1..3}|xargs -n1|sed -E 's#(.*)#useradd \1;p=`mkpasswd`;echo $p|passwd --stdin \1;echo \1 " : " $p >> password.txt#g'
```

2、权限管理

```bash
1.rwx 数字表示
	（1）r ==== 4
	（2）w ==== 2
	（3）x ==== 1

2.权限表示方式
	（1）每三位换算成数字相加
		rw-r--r--   644
		属主：rw-   4+2+0
		属组：r--   4+0+0
		其他：r--   4+0+0

3.修改文件权限
	（1）使用rwx修改文件权限
		a.增加
		chmod ugo+rwx file
		b.减少
		chmod ugo-rwx file
		c.等于
		chmod ugo=rwx file   #清空原有权限，重新赋值
		d.不写ugo身份，rx修改全部人，w修改属主
		chmod +/-rwx file
		
		ugo身份：
		u   属主 user
		g   属组 group
		o   其他 others
	
		示例：
			#授权属主x权限
			chmod u+x a.txt
			
			#剥夺属主w权限
			chmod u-w a.txt
			
			#授权属组wx权限
			chmod g+wx a.txt
			
			#剥夺属组全部权限
			chmod g-rwx a.txt
			
			#授权属组和其他x权限
			chmod go+x a.txt
	
	（2）使用数字授予权限
		chmod 744 a.txt  #授予rwxr--r--
```

3、对于文件rwx的含义

```bash

情况一：是属主，但没有wx权限，可以使用vim和:wq!强制修改
	结果：
		可以看，可以写，不能执行
	[bob@breadbomb bob]$ ll 1.txt
	-r--r--r-- 1 bob bob 0 May 19 11:23 1.txt
	
	[bob@breadbomb bob]$ vim 1.txt

情况二：是属主，没rx权限，但有w权限，可以使用echo >> file
	结果：
		不能看，只能使用> >>追加，不可执行
	[bob@breadbomb bob]$ ll 1.txt
	--w-r--r-- 1 bob bob 6 May 19 11:35 1.txt
	
	[bob@breadbomb bob]$ echo hello >> 1.txt

情况三：是属主，没rw权限，但有x权限
	结果：
		什么都不能干
	[bob@breadbomb bob]$ ll 1.txt
	---xr--r-- 1 bob bob 12 May 19 11:39 1.txt

总结：
	（1）可以查看文件，必须有 r 权限
	（2）可以读写文件，必须有 rw 权限
	（3）可以执行文件，必须有 rx 权限
	（4）文件权限只控制文件的可读、可写、可执行，删除不了（删除由目录权限控制）

普通文件的最高权限：666
企业中所有普通文件权限最高为：644
```
