
1、批量创建用户

```bash
1.带空格拼接
	[root@breadbomb ~]# echo 'useradd bob'{1..3}|xargs -n 2
	useradd bob1
	useradd bob2
	useradd bob3

2.分隔符拼接
	[root@breadbomb ~]# echo 'useradd_bob'{1..3}|sed 's#_# #g'|xargs -n 2
	useradd bob1
	useradd bob2
	useradd bob3

3.后向引用拼接
	[root@breadbomb ~]# echo 'bob'{1..3}|xargs -n 1|sed -E 's#(.*)#useradd \1#g'
	useradd bob1
	useradd bob2
	useradd bob3

#以上最后统一加上 |bash 即可执行
```
