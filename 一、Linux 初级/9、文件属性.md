
1、文件属性

```bash
1.ls 命令
	-l 详细信息 ll
	-a 显示隐藏文件
	-i 显示inode号码
	-r 逆序显示文件
	-t 按照文件创建时间显示
	-d 目录本身

[root@breadbomb ~]# ll -i example
total 4
51262230 -rw-r--r-- 1 root root 12 May 14 09:54 a.txt

第一列：inode号码
	作用：通过文件名找到inode号码，再通过号码找到磁盘空间真正存储的位置
	
	inode号码有数量上限，代表文件上限
		可通过df -i 查看磁盘的inode数量上限
		
		[root@breadbomb ~]# df -i
		Filesystem      Inodes IUsed   IFree IUse% Mounted on
		devtmpfs        250803   378  250425    1% /dev
		tmpfs           253483     1  253482    1% /dev/shm
		tmpfs           253483   719  252764    1% /run
		tmpfs           253483    16  253467    1% /sys/fs/cgroup
		/dev/sda3      9334272 60886 9273386    1% /
		/dev/sda1       102400   326  102074    1% /boot
		tmpfs           253483     1  253482    1% /run/user/0
	
		面试题：为什么磁盘空间没法创建新文件了？
		a.磁盘空间损坏
		b.inode号码上限，即使空间占用小也无法继续创建
		c.磁盘空间已满

第二列： 文件类型
	
	文件类型：
		a.- 普通文件
		b.d 目录
		c.l 软链接 表示快捷方式
		d.b 块设备 表示硬件
		e.c 字符设备
			[root@breadbomb ~]# ll /dev/urandom
			crw-rw-rw- 1 root root 1, 9 May 13 11:56 /dev/urandom
			[root@breadbomb ~]# ll /dev/zero
			crw-rw-rw- 1 root root 1, 5 May 13 11:56 /dev/zero
			[root@breadbomb ~]# ll /dev/null
			crw-rw-rw- 1 root root 1, 3 May 13 11:56 /dev/null
			
			urandom 文件：
				查看该文件，一直输出乱码
			zero 文件：
				生成定量数据
				[root@breadbomb ~]# dd if=/dev/zero of=10g.txt bs=1M count=10000
				解析：dd生成 if输入源 of输出地 bs每次读写1M count总次数 ，即生成一个10g文件
				
			null 文件：
				黑洞 直接销毁，不可还原 大小永远为0
				
				#案例：有时候指令输出结果只需知道成功还是失败
				
				$?：表示上一条指令结果是否成功 成功：0  不成功：非0
				
					[root@breadbomb ~]# ping -W1 -c1 baidu.com
					成功结果省略
					[root@breadbomb ~]# echo $?
					0
					
					[root@breadbomb ~]# ping baiduuuuuu.com
					ping: baiduuuuuu.com: Name or service not known
					[root@breadbomb ~]# echo $?
					2
					
					
					[root@breadbomb ~]# cat example/a.txt 
					ping -W1 -c1 www.baidu.com &>/dev/null  （&> 接受正确、错误）
					if [$? -eq 0];then
						echo "www.baidu.com 能ping通"
					else
						echo "www.baidu.com 不能ping通"
					fi
					
				&>: 表示正确错误都覆盖到文件中
				&>>:追加到文件中
					
					echo hello &> a.txt
					等价于
					echo hello > a.txt 2> a.txt
					等价于
					echo hello > a.txt 2>&1 (固定格式，代表而此操作和1一样)
					
					echo hello &>> a.txt
					等价于
					echo hello >> a.txt 2>> a.txt
					等价于
					echo hello >> a.txt 2>&1 (固定格式，代表二次操作和1一样) 
```


2、软件安装

```bash
安装方式：
	1.yum 安装
	2.rpm 安装
	3.编译安装
	
	yum安装
		
		a.安装
		yum install 软件名                     # 安装指定软件
		yum groupinstall "Development Tools"  # 安装软件组
		
		b.查询
		yum search 关键词                 # 搜索包
		yum info 软件名                   # 查看包详情
		yum list                         # 列出已安装/可安装包
		yum grouplist                    # 查看软件组
		
		c.更新
		yum update                       # 更新所有可更新包
		yum update 软件名                 # 更新指定软件
		
		d.卸载
		yum remove 软件名                 # 卸载软件（保留配置）
		yum erase 软件名                  # 彻底删除（含配置）
		
		e.仓库与缓存
		yum repolist                     # 查看已启用仓库
		yum clean all                    # 清空缓存（元数据+包）
		
		f.只下载不安装、指定路径
		默认路径：/var/cache/yum/x86_64/7/仓库名/packages/
		yum install -y --downloadonly --downloaddir=/usr/local/rpm nginx
		
			windows和linux传输
			（1）yum -y install lrzsz
			（2）该包中有两个命令
				rz:将windows中的文件上传到linux
					rz默认上传到当前目录下
				sz:将linux中的文件下载到windows
					sz需要打包压缩才能下载目录，正常只能下载文件
	
	rpm 安装
		rpm负责安装本地已下载好的.rpm软件包
		特点：不支持解决依赖包问题
		-i 安装
		-v 显示详细过程
		-h 显示进度条
		
		-qa 查看已安装软件
			rpm -qa | grep cow
			rpm -qa cowsay
		-ql 查看某个包的所有路径
			rpm -ql cowsay
		
		卸载：
		-e 卸载 ，最好是用yum remove , rpm -e 不支持依赖
		
		rpm -ivh 软件包名.rpm
	
```

4、压缩和解压

```bash
tar 命令 压缩和解压
	-z 用gzip压缩
	-c 创建压缩包
	-x 解压
	-v 展示过程
	-f 指定压缩包文件名
	-C 压缩、解压到指定位置
	
		# 压缩整个文件夹 为 文件夹名.tar.gz
		tar -zcvf 压缩包名.tar.gz 要压缩的文件夹/
		
		# 示例：把 test 文件夹压缩成 test.tar.gz
		tar -zcvf test.tar.gz test/
		
		# 解压到当前目录
		tar -zxvf test.tar.gz
		
		# 解压到指定目录
		tar -zxvf test.tar.gz -C /opt/
		
zip 和 unzip 命令（需要安装）
	yum -y install zip unzip
	
	zip 压缩
		-r 递归压缩
		
		#压缩test目录
		zip -r test.zip test/ 
		
	unzip 解压
		-d 指定目录
		unzip test.zip 
		
		# 解压到指定目录 
		unzip test.zip -d /opt/
```
