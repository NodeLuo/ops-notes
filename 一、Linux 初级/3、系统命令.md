
1、关机重启命令

```bash
1.关机
	shutdown   关机 默认1分钟
		-h
		shutdown -h 5    5分钟后关机
		shutdown -h now  立刻关机
	
	poweroff   关机
	
	init 0     关机


2.重启
	shutdown -r 重启
		-r
		shutdown -r 5    5分钟后重启
		shutdown -r now  立刻重启
	
	reboot      重启
	
	init 6      重启
```

2、基础命令

```bash
1.pwd 查看当前路径

[root@breadbomb ~]# pwd
/root

2.cd 切换目录
	a.回家
		cd ~
		cd /root
		cd 直接回车
	b.回到上一次所在目录
		cd -
	c.上一级目录
		cd ..

3.ls 查看文件
	-l 查看详细信息
	-a 查看所有文件，包括隐藏文件 （.file）
	
	可以使用alias命令查看别名
	ll ----> ls -l 简写

4.touch 创建文件 若文件存在，则只修改文件时间
	touch file1 file2
	touch /root/file1
	注意：该命令不可递归创建文件，只能在已有目录下创建

5.mkdir 创建目录
	mkdir dir1
	mkdir dir1 dir2
	-p 递归创建

6.tree 查看当前目录里面的树结构
	-L 1 只显示一级目录
	
	需要安装
	yum -y install tree

7.cat 查看文件内容
	-n 显示行号
	
	合并文件：
	cat file1 file2 > file3
```
